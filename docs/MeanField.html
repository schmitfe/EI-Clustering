MeanField

Mean-field solvers and EI-cluster specializations.

The package exposes a generic fixed-point solver in RateSystem and a concrete clustered E/I specialization in EIClusterNetwork.

Minimal example:

from MeanField import EIClusterNetwork

system = EIClusterNetwork(parameter, v_focus=0.2)
x, residual, success = system.solve()
rates = system.full_rates_numpy(x)

The helpers in rate_system are used by the figure pipelines to trace event-rate functions, store fixpoint bundles, and reuse cached results.

143    def _build_population_groups(self, focus: np.ndarray) -> List[np.ndarray]:
144        focus_set = set(int(idx) for idx in focus.tolist())
145        if not focus_set:
146            focus_set = {0}
147        groups: List[np.ndarray] = [np.array(sorted(focus_set), dtype=int)]
148        remaining = [idx for idx in range(self.population_count) if idx not in focus_set]
149        for idx in remaining:
150            groups.append(np.array([idx], dtype=int))
151        return groups

154    def _initialize_group_constraints(self) -> None:
155        groups = self._build_population_groups(self.focus_indices)
156        covered = np.concatenate(groups)
157        if covered.size != self.population_count or not np.all(np.sort(covered) == np.arange(self.population_count)):
158            raise ValueError("Population groups must cover each population exactly once.")
159        self.groups: List[np.ndarray] = groups
160        self.group_count = len(groups)
161        self.focus_group_index = 0
162        self.solve_groups = [idx for idx in range(self.group_count) if idx != self.focus_group_index]
163        self.dim = len(self.solve_groups)
164        self.focus_population_mask = np.zeros(self.population_count, dtype=bool)
165        self.focus_population_mask[self.focus_indices] = True
166        self.group_membership = np.zeros((self.population_count, self.group_count), dtype=float)
167        for idx, members in enumerate(self.groups):
168            self.group_membership[members, idx] = 1.0
169        self.group_sizes = np.array([len(members) for members in self.groups], dtype=float)
170        self.group_inverse_sizes = np.reciprocal(self.group_sizes)
171        self.focus_vector = np.zeros(self.group_count, dtype=float)
172        self.focus_vector[self.focus_group_index] = 1.0
173        self.selector_matrix = np.zeros((self.group_count, self.dim), dtype=float)
174        for col, group_idx in enumerate(self.solve_groups):
175            self.selector_matrix[group_idx, col] = 1.0
176        self.residual_matrix = np.zeros((self.dim, self.population_count), dtype=float)
177        for row, group_idx in enumerate(self.solve_groups):
178            members = self.groups[group_idx]
179            self.residual_matrix[row, members] = 1.0 / len(members)

181    def _resolve_focus_indices(self, focus_config) -> np.ndarray:
182        if focus_config is None:
183            entries = [0]
184        elif isinstance(focus_config, (int, np.integer)):
185            entries = [int(focus_config)]
186        else:
187            entries = []
188            for value in focus_config:
189                if isinstance(value, slice):
190                    start = 0 if value.start is None else value.start
191                    stop = self.population_count if value.stop is None else value.stop
192                    step = 1 if value.step is None else value.step
193                    entries.extend(range(start, stop, step))
194                else:
195                    entries.append(int(value))
196        focus = sorted(set(entries))
197        if not focus:
198            focus = [0]
199        for idx in focus:
200            if idx < 0 or idx >= self.population_count:
201                raise ValueError(f"Focus index {idx} out of bounds for population size {self.population_count}.")
202        return np.asarray(focus, dtype=int)

204    def _full_rates_numpy(self, x: np.ndarray) -> np.ndarray:
205        arr = np.asarray(x, dtype=float).ravel()
206        if arr.size == self.population_count:
207            return arr
208        if arr.size == self.group_count:
209            group_values = arr.copy()
210            group_values[self.focus_group_index] = self.v_focus
211        else:
212            if arr.size != self.dim:
213                raise ValueError(
214                    f"Expected vector of length {self.dim}, {self.group_count}, or {self.population_count}, got {arr.size}."
215                )
216            group_values = self.selector_matrix @ arr
217            group_values += self.focus_vector * self.v_focus
218        return self.group_membership @ group_values

220    def _phi_numpy(self, full_rates: np.ndarray) -> np.ndarray:
221        mean = self.A.dot(full_rates) + self.bias
222        rates_sq = full_rates * full_rates
223        var = np.maximum(self.B.dot(full_rates) + self.C.dot(rates_sq), self.VAR_EPS)
224        return 0.5 * (1 - special.erf(-mean / np.sqrt(2.0 * var)))
225
226    def _phi_jacobian_numpy(self, full_rates: np.ndarray) -> np.ndarray:
227        mean = self.A.dot(full_rates) + self.bias
228        rates_sq = full_rates * full_rates
229        var = np.maximum(self.B.dot(full_rates) + self.C.dot(rates_sq), self.VAR_EPS)
230        inv_sqrt = 1.0 / np.sqrt(2.0 * var)
231        exp_term = np.exp(-(mean ** 2) / (2.0 * var))
232        coeff = (1.0 / np.sqrt(np.pi)) * exp_term * inv_sqrt
233        correction = mean / (2.0 * var)
234        dvar_dm = self.B + self.C * (2.0 * full_rates[None, :])
235        return coeff[:, None] * (self.A - correction[:, None] * dvar_dm)

237    def residual_numpy(self, x: np.ndarray) -> np.ndarray:
238        rates = self._full_rates_numpy(x)
239        phi = self._phi_numpy(rates)
240        residual = (phi - rates) / self.tau
241        if self.dim == 0:
242            return np.zeros((0,), dtype=float)
243        return self.residual_matrix @ residual

245    @staticmethod
246    def _jax_residual(x, v_focus, args):
247        A, B, C, bias, tau, focus_vector, selector, membership, reduction, var_eps = args
248        group_values = selector @ x + focus_vector * v_focus
249        full_rates = membership @ group_values
250        mean = A @ full_rates + bias
251        var = jnp.maximum(B @ full_rates + C @ (full_rates * full_rates), var_eps)
252        phi = 0.5 * (1 - jspecial.erf(-mean / jnp.sqrt(2.0 * var)))
253        residual = (phi - full_rates) / tau
254        return reduction @ residual

256    def solve(self, initial_guess: Optional[np.ndarray] = None) -> Tuple[np.ndarray, np.ndarray, bool]:
257        """Solve for a fixed point of the reduced mean-field system.
258
259        Returns
260        -------
261        tuple[np.ndarray, np.ndarray, bool]
262            `(x, residual, success)` where `x` is the reduced solution vector.
263
264        Expected output
265        ---------------
266        `success` is `True` when the nonlinear solver converged and `residual`
267        is close to zero.
268        """
269        initial = self._coerce_initial(initial_guess)
270        if self.dim == 0:
271            residual = self.residual_numpy(initial)
272            return initial, residual, True
273        if self.use_jax:
274            try:
275                return self._solve_with_optimistix(initial)
276            except SolverConvergenceError:
277                raise
278            except (ValueError, RuntimeError, TypeError, AttributeError) as e:
279                logger.warning(
280                    "Optimistix solver failed with %s: %s. Falling back to scipy solver.",
281                    type(e).__name__,
282                    str(e),
283                )
284                self.use_jax = False
285        return self._solve_with_scipy(initial)

297    def _solve_with_optimistix(self, initial: np.ndarray) -> Tuple[np.ndarray, np.ndarray, bool]:
298        args = self._prepare_jax_args()
299        solver_entry = self._get_jax_solver()
300        if solver_entry is None:
301            raise RuntimeError("Optimistix solver is unavailable.")
302        x0 = jnp.asarray(initial, dtype=jnp.float64)
303        v_focus = jnp.asarray(float(self.v_focus), dtype=jnp.float64)
304        value, status, _ = solver_entry["single"](x0, v_focus, args)
305        success = bool(np.asarray(status, dtype=bool))
306        if not success:
307            raise SolverConvergenceError(self.v_focus, self.max_steps)
308        value_np = np.asarray(value, dtype=float)
309        return value_np, self.residual_numpy(value_np), success

373    def _reduce_full_rates(self, full_rates: np.ndarray) -> np.ndarray:
374        if self.dim == 0:
375            return np.zeros((0,), dtype=float)
376        full = np.asarray(full_rates, dtype=float).reshape((self.population_count,))
377        group_sums = self.group_membership.T @ full
378        group_means = group_sums * self.group_inverse_sizes
379        return group_means[self.solve_groups]

381    def _coerce_initial(self, initial_guess: Optional[np.ndarray]) -> np.ndarray:
382        if self.dim == 0:
383            return np.zeros((0,), dtype=float)
384        if initial_guess is None:
385            return np.full((self.dim,), 0.1, dtype=float)
386        arr = np.asarray(initial_guess, dtype=float).ravel()
387        if arr.size == self.dim:
388            return arr
389        if arr.size == self.group_count:
390            copy = arr.copy()
391            copy[self.focus_group_index] = self.v_focus
392            return copy[self.solve_groups]
393        if arr.size == self.population_count:
394            return self._reduce_full_rates(arr)
395        non_focus_count = self.population_count - len(self.focus_indices)
396        if arr.size == non_focus_count:
397            full_rates = np.empty(self.population_count, dtype=float)
398            full_rates[self.focus_population_mask] = self.v_focus
399            full_rates[~self.focus_population_mask] = arr
400            return self._reduce_full_rates(full_rates)
401        raise ValueError(
402            f"Initial guess must have length {self.dim}, {self.group_count}, {self.population_count}, or {non_focus_count}, got {arr.size}."
403        )

405    def solve_sequence(
406        self,
407        v_focus_values: Sequence[float],
408        initial_guess: Optional[np.ndarray] = None,
409    ) -> Optional[Tuple[np.ndarray, np.ndarray]]:
410        """Solve a sequence of focus inputs with the accelerated JAX path.
411
412        Expected output
413        ---------------
414        Returns `(solutions, success_flags)` when the JAX solver is available,
415        otherwise `None`.
416        """
417        values = np.asarray(list(v_focus_values), dtype=float)
418        if values.size == 0:
419            return np.zeros((0, self.dim)), np.zeros((0,), dtype=bool)
420        if self.dim == 0:
421            zeros = np.zeros((values.size, 0), dtype=float)
422            return zeros, np.ones((values.size,), dtype=bool)
423        if not (self.use_jax and HAS_JAX and optx is not None):
424            return None
425        solver_entry = self._get_jax_solver()
426        if solver_entry is None:
427            return None
428        args = self._prepare_jax_args()
429        initial = self._coerce_initial(initial_guess)
430        x0 = jnp.asarray(initial, dtype=jnp.float64)
431        v_seq = jnp.asarray(values, dtype=jnp.float64)
432        try:
433            solutions, statuses, _ = solver_entry["scan"](x0, v_seq, args)
434        except (ValueError, RuntimeError, TypeError, AttributeError) as exc:
435            logger.warning(
436                "Optimistix sweep failed with %s: %s. Falling back to sequential solver.",
437                type(exc).__name__,
438                str(exc),
439            )
440            self.use_jax = False
441            return None
442        solution_np = np.asarray(solutions, dtype=float)
443        success = np.asarray(statuses, dtype=bool)
444        return solution_np, success

448    def generate_erf_curve(
449        cls,
450        parameter: Dict,
451        *,
452        start: float = 0.02,
453        end: float = 1.0,
454        step_number: int = 20,
455        retry_step: Optional[float] = None,
456        initial_guess: Optional[np.ndarray] = None,
457        fallback_initials: Optional[Sequence[float]] = None,
458        **network_kwargs,
459    ) -> ERFResult:
460        """Compute the ERF for ``parameter``.
461
462        When ``retry_step`` is provided, the solver retries inputs separated by
463        that value whenever convergence fails. If the ERF cannot be completed
464        the ``completed`` flag is ``False`` and no serialization should happen.
465        Parameters
466        ----------
467        start : float, optional
468            Lower bound of the input range for the ERF (default: 0.02).
469        end : float, optional
470            Upper bound of the input range for the ERF (default: 1.0).
471        step_number : int, optional
472            Number of steps between ``start`` and ``end`` (default: 20).
473        ...
474
475        Expected output
476        ---------------
477        Returns an :class:`ERFResult` whose `x_data` stores the driven focus
478        inputs and whose `y_data` stores the corresponding mean-field outputs.
479        """
480        ERF_EPS = 1e-12  # Tolerance for floating-point comparison of v_in against end
481        x_data: List[float] = []
482        y_data: List[float] = []
483        solves: List[np.ndarray] = []
484        step = (end - start) / max(step_number, 1)
485        aborted = False
486        fallback_values = list(fallback_initials) if fallback_initials is not None else [0.02, 0.2, 0.5, 0.8, 0.98]
487        vector_values: Optional[List[float]] = None
488        if retry_step is None:
489            vector_values = []
490            if step == 0:
491                vector_values.append(float(start))
492            else:
493                cursor = float(start)
494                while cursor <= end + ERF_EPS:
495                    vector_values.append(cursor)
496                    cursor += step
497        initial_focus = float(start)
498        if vector_values:
499            initial_focus = float(vector_values[0])
500        system = cls(parameter, initial_focus, **network_kwargs)
501        current_initial = initial_guess
502        prefetched_solutions: Optional[np.ndarray] = None
503        prefix_limit = 0
504        if vector_values and system.use_jax:
505            seq_result = system.solve_sequence(vector_values, initial_guess=current_initial)
506            if seq_result is not None:
507                prefetched_solutions, success_flags = seq_result
508                failure = np.flatnonzero(~success_flags)
509                prefix_limit = int(failure[0]) if failure.size else len(vector_values)
510        if vector_values is not None:
511            total_points = len(vector_values)
512            idx = 0
513            while idx < total_points:
514                v_in = vector_values[idx]
515                system.v_focus = float(v_in)
516                use_prefetched = prefetched_solutions is not None and idx < prefix_limit
517                if use_prefetched:
518                    solution = np.asarray(prefetched_solutions[idx], dtype=float)
519                    success = True
520                else:
521                    skip_due_to_solver = False
522                    try:
523                        solution, residual, success = system.solve(current_initial)
524                    except SolverConvergenceError as exc:
525                        logger.debug("Skipping v_in %.6f: %s", float(v_in), str(exc))
526                        skip_due_to_solver = True
527                    if skip_due_to_solver:
528                        idx += 1
529                        continue
530                    if not success:
531                        for seed in fallback_values:
532                            if system.dim == 0:
533                                candidate = np.zeros((0,), dtype=float)
534                            else:
535                                candidate = np.full((system.dim,), float(seed), dtype=float)
536                            try:
537                                solution, residual, success = system.solve(candidate)
538                            except SolverConvergenceError as exc:
539                                logger.debug("Skipping v_in %.6f: %s", float(v_in), str(exc))
540                                skip_due_to_solver = True
541                                break
542                            if success:
543                                break
544                        if skip_due_to_solver:
545                            idx += 1
546                            continue
547                        if not success:
548                            aborted = True
549                            break
550                phi_values = system.phi_numpy(solution)
551                x_data.append(system.v_focus)
552                y_data.append(system.focus_output(phi_values))
553                solves.append(solution)
554                current_initial = solution
