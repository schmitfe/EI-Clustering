  43@njit(cache=True)
  44def _dense_batch_kernel(
  45    neurons,
  46    state,
  47    field,
  48    thresholds,
  49    weights,
  50    log_states,
  51    log_enabled,
  52    log_offset,
  53    update_log,
  54    delta_log,
  55    diff_enabled,
  56    diff_offset,
  57):
  58    """
  59    Update neurons in-place for dense weight matrices.
  60    Parameters
  61    ----------
  62    neurons : np.ndarray
  63        Indices that should be updated (int64).
  64    state : np.ndarray
  65        Binary neuron state vector (int8).
  66    field : np.ndarray
  67        Cached synaptic field h = W @ state (float32).
  68    thresholds : np.ndarray
  69        Per-neuron thresholds (float32).
  70    weights : np.ndarray
  71        Dense weight matrix (float32).
  72    """
  73    neuron_count = weights.shape[0]
  74    for idx in range(neurons.size):
  75        neuron = neurons[idx]
  76        old_state = state[neuron]
  77        potential = field[neuron]
  78        new_state = 1 if potential > thresholds[neuron] else 0
  79        delta = 0
  80        if new_state != old_state:
  81            delta = new_state - old_state
  82            state[neuron] = new_state
  83            for target in range(neuron_count):
  84                field[target] += delta * weights[target, neuron]
  85        if diff_enabled:
  86            update_log[diff_offset + idx] = neuron
  87            delta_log[diff_offset + idx] = delta
  88        if log_enabled:
  89            log_states[log_offset + idx, :] = state

 137@njit(cache=True)
 138def _reconstruct_states_kernel(initial_state, updates, deltas, stride):
 139    steps = updates.shape[1]
 140    if steps == 0:
 141        return np.zeros((0, initial_state.size), dtype=np.uint8)
 142    sample_count = ((steps - 1) // stride) + 1
 143    state = initial_state.copy()
 144    samples = np.zeros((sample_count, state.size), dtype=np.uint8)
 145    sample_idx = 0
 146    for step in range(steps):
 147        for row in range(updates.shape[0]):
 148            delta = deltas[row, step]
 149            if delta == 0:
 150                continue
 151            unit = updates[row, step]
 152            state_value = int(state[unit]) + int(delta)
 153            state[unit] = 1 if state_value > 0 else 0
 154        if step % stride == 0:
 155            samples[sample_idx, :] = state
 156            sample_idx += 1
 157    return samples

 160@njit(cache=True)
 161def _population_rates_kernel(cluster_sums, sizes, cluster_of, updates, deltas, stride):
 162    steps = updates.shape[1]
 163    pop_count = cluster_sums.size
 164    if steps == 0 or pop_count == 0:
 165        return np.zeros((0, pop_count), dtype=np.float32)
 166    sample_count = ((steps - 1) // stride) + 1
 167    samples = np.zeros((sample_count, pop_count), dtype=np.float32)
 168    running = cluster_sums.copy()
 169    sample_idx = 0
 170    for step in range(steps):
 171        for row in range(updates.shape[0]):
 172            delta = deltas[row, step]
 173            if delta == 0:
 174                continue
 175            unit = updates[row, step]
 176            cluster_idx = cluster_of[unit]
 177            running[cluster_idx] += int(delta)
 178        if step % stride == 0:
 179            for pop_idx in range(pop_count):
 180                samples[sample_idx, pop_idx] = running[pop_idx] / sizes[pop_idx]
 181            sample_idx += 1
 182    return samples

 185def warm_numba_caches() -> None:

 187    global _NUMBA_WARMED
 188    if _NUMBA_WARMED:

 245def _compress_flat_samples(flat: np.ndarray, total_pairs: int) -> tuple[np.ndarray, np.ndarray]:
 246    flat = np.asarray(flat, dtype=np.int64).ravel()
 247    if flat.size == 0:
 248        return np.zeros(0, dtype=np.int64), np.zeros(0, dtype=np.int64)
 249    if total_pairs <= 2_000_000:
 250        counts = np.bincount(flat, minlength=total_pairs)
 251        unique = np.flatnonzero(counts)
 252        return unique.astype(np.int64, copy=False), counts[unique].astype(np.int64, copy=False)
 253    unique, counts = np.unique(flat, return_counts=True)
 254    return unique.astype(np.int64, copy=False), counts.astype(np.int64, copy=False)

 290class BinaryNeuronPopulation(Neuron):
 291    """Binary neuron population with threshold activation and configurable initializer.
 292
 293    Examples
 294    --------
 295    >>> net = BinaryNetwork("demo")
 296    >>> pop = BinaryNeuronPopulation(net, N=3, threshold=0.5, initializer=0)
 297    >>> net.add_population(pop)
 298    <...BinaryNeuronPopulation object...>
 299    """
 300    def __init__(
 301        self,
 302        reference,
 303        N=1,
 304        threshold=1.0,
 305        name="Binary Neuron Population",
 306        tau=1.0,
 307        initializer=None,
 308        **kwargs,
 309    ):
 310        super().__init__(reference, N, name, tau=tau)
 311        self.threshold = float(threshold)
 312        self.initializer = initializer
 313
 314    def update(self, weights=None, state=None, index=None, input_value=None, **kwargs):
 315        if input_value is None:
 316            if weights is None or state is None:
 317                raise ValueError("weights and state have to be provided when input_value is not set")
 318            input_value = np.sum(weights * state)
 319        return np.heaviside(input_value - self.threshold, 0)
 320
 321    def _initial_state(self):
 322        if callable(self.initializer):
 323            values = np.asarray(self.initializer(self.N), dtype=np.int16)
 324            if values.size != self.N:
 325                raise ValueError("Initializer must return {} entries".format(self.N))
 326            return values
 327        if self.initializer is None:
 328            return np.random.choice([0, 1], size=self.N, p=[0.8, 0.2])
 329        arr = np.asarray(self.initializer)
 330        if arr.size == 1:
 331            return np.full(self.N, int(arr.item()), dtype=np.int16)
 332        if arr.size != self.N:
 333            raise ValueError("Initializer must define exactly {} elements".format(self.N))
 334        return arr.astype(np.int16)

 337class BackgroundActivity(Neuron):
 338    """Background population that provides a constant or stochastic drive."""
 339    def __init__(self, reference, N=1, Activity=0.5, Stochastic=False, name="Background Activity", tau=1.0):
 340        super().__init__(reference, N, name, tau=tau)
 341        self.Activity = Activity
 342        if Stochastic:
 343            self.update = self.update_stochastic
 344        else:
 345            self.update = self.update_deterministic
 346
 347    def update_stochastic(self, weights=None, state=None, Index=None, **kwargs):
 348        return np.random.choice([0, 1], 1) * self.update_deterministic(weights, state, **kwargs)
 349
 350    def update_deterministic(self, weights=None, state=None, Index=None, **kwargs):
 351        # if activity is a float, set all neurons to this activity
 352        if isinstance(self.Activity, float):
 353            return self.Activity
 354        # if activity is a function, set neurons by this function
 355        elif callable(self.Activity):
 356            return self.Activity()
 357        else:
 358            return 1.0
 359
 360    def initialze(self):
 361        self.state = np.array([self.update() for _ in range(self.N)])

 364class Synapse(NetworkElement):
 365    """Base class for synapse objects that write weight blocks into the network."""
 366    def __init__(self, reference, pre, post, name="Some Synapse"):
 367        super().__init__(reference, name=post.name + " <- " + pre.name)
 368        self.pre = pre
 369        self.post = post
 370
 371    def set_view(self, view):
 372        super().set_view(view)
 373
 374    def _write_block(self, block: np.ndarray):
 375        """Add a dense block to the global weight matrix (dense or sparse backend)."""
 376        post_start, post_end = int(self.view[1, 0]), int(self.view[1, 1])
 377        pre_start, pre_end = int(self.view[0, 0]), int(self.view[0, 1])
 378        self.reference._write_weight_block(post_start, post_end, pre_start, pre_end, block)
 379
 380    def _append_sparse_entries(self, rows: np.ndarray, cols: np.ndarray, values: np.ndarray) -> None:
 381        if self.reference.weight_mode != "sparse":
 382            raise RuntimeError("Sparse entries can only be appended in sparse weight mode.")
 383        if rows.size == 0:
 384            return
 385        post_start = int(self.view[1, 0])
 386        pre_start = int(self.view[0, 0])
 387        self.reference._append_sparse_entries(
 388            np.asarray(rows, dtype=np.int64) + post_start,
 389            np.asarray(cols, dtype=np.int64) + pre_start,
 390            np.asarray(values, dtype=self.reference.weight_dtype),
 391        )
 392
 393    def initialze(self):
 394        raise NotImplementedError

 397class PairwiseBernoulliSynapse(Synapse):
 398    """Sample independent Bernoulli connections for every pre/post pair.
 399
 400    Expected output
 401    ---------------
 402    After `network.initialize(...)`, the addressed weight block contains zeros
 403    and `j`-valued entries sampled independently with probability `p`.
 404    """
 405    def __init__(self, reference, pre, post, p=0.5, j=1.0):
 406        super().__init__(reference, pre, post)
 407        self.p = float(p)
 408        self.j = float(j)
 409
 410    def initialze(self):
 411        p = self.p
 412        iterations = 1
 413        while p > 1:
 414            p /= 2.0
 415            iterations += 1
 416        if self.reference.weight_mode == "sparse":
 417            total_pairs = self.post.N * self.pre.N
 418            if total_pairs == 0 or p <= 0.0:
 419                return
 420            data_chunks: List[np.ndarray] = []
 421            row_chunks: List[np.ndarray] = []
 422            col_chunks: List[np.ndarray] = []
 423            for _ in range(iterations):
 424                sample_count = int(np.random.binomial(total_pairs, p))
 425                if sample_count <= 0:
 426                    continue
 427                if sample_count >= total_pairs:
 428                    flat = np.arange(total_pairs, dtype=np.int64)
 429                else:
 430                    flat = np.random.choice(total_pairs, size=sample_count, replace=False).astype(np.int64, copy=False)
 431                rows = flat // self.pre.N
 432                cols = flat % self.pre.N
 433                row_chunks.append(rows)
 434                col_chunks.append(cols)
 435                data_chunks.append(np.full(sample_count, self.j, dtype=self.reference.weight_dtype))
 436            if data_chunks:
 437                self._append_sparse_entries(
 438                    np.concatenate(row_chunks),
 439                    np.concatenate(col_chunks),
 440                    np.concatenate(data_chunks),
 441                )
 442            return
 443        shape = (self.post.N, self.pre.N)
 444        block = np.zeros(shape, dtype=self.reference.weight_dtype)
 445        for _ in range(iterations):
 446            draws = (np.random.random(size=shape) < p).astype(block.dtype, copy=False)
 447            block += draws * self.j
 448        self._write_block(block)

 451class PoissonSynapse(Synapse):
 452    """Sample Poisson-distributed multi-edge counts per pre/post pair.
 453
 454    Expected output
 455    ---------------
 456    The addressed weight block contains non-negative multiples of `j` with
 457    Poisson-distributed counts.
 458    """
 459    def __init__(self, reference, pre, post, rate=0.5, j=1.0):
 460        super().__init__(reference, pre, post)
 461        self.rate = float(rate)
 462        self.j = float(j)
 463
 464    def initialze(self):
 465        if self.reference.weight_mode == "sparse":
 466            total_pairs = self.post.N * self.pre.N
 467            if total_pairs == 0 or self.rate <= 0.0:
 468                return
 469            sample_count = int(np.random.poisson(lam=self.rate * total_pairs))
 470            if sample_count <= 0:
 471                return
 472            flat = np.random.randint(total_pairs, size=sample_count, dtype=np.int64)
 473            unique, counts = _compress_flat_samples(flat, total_pairs)
 474            rows = unique // self.pre.N
 475            cols = unique % self.pre.N
 476            values = counts.astype(self.reference.weight_dtype, copy=False) * self.j
 477            self._append_sparse_entries(rows, cols, values)
 478            return
 479        shape = (self.post.N, self.pre.N)
 480        samples = np.random.poisson(lam=self.rate, size=shape).astype(self.reference.weight_dtype, copy=False)
 481        self._write_block(samples * self.j)

 484class FixedIndegreeSynapse(Synapse):
 485    """Connect each target neuron to a fixed number of randomly drawn inputs.
 486
 487    Expected output
 488    ---------------
 489    Each target row receives approximately `round(p * N_pre)` incoming entries.
 490    """
 491    def __init__(self, reference, pre, post, p=0.5, j=1.0):
 492        super().__init__(reference, pre, post)
 493        self.p = float(p)
 494        self.j = float(j)
 495
 496    def initialze(self):
 497        p = max(self.p, 0.0)
 498        target_count = int(round(p * self.pre.N))
 499        target_count = min(max(target_count, 0), self.pre.N)
 500        if target_count == 0:
 501            return
 502        if self.reference.weight_mode == "sparse":
 503            rows = np.repeat(np.arange(self.post.N, dtype=np.int64), target_count)
 504            cols = np.random.randint(self.pre.N, size=self.post.N * target_count, dtype=np.int64)
 505            values = np.full(rows.size, self.j, dtype=self.reference.weight_dtype)
 506            self._append_sparse_entries(rows, cols, values)
 507            return
 508        block = np.zeros((self.post.N, self.pre.N), dtype=self.reference.weight_dtype)
 509        for tgt in range(self.post.N):
 510            pres = np.random.choice(self.pre.N, size=target_count, replace=True)
 511            np.add.at(block[tgt], pres, self.j)
 512        self._write_block(block)

 515class AllToAllSynapse(Synapse):
 516    """Dense all-to-all block with constant weight value.
 517
 518    Expected output
 519    ---------------
 520    The addressed weight block is filled with the constant value `j`.
 521    """
 522    def __init__(self, reference, pre, post, j=1.0):
 523        super().__init__(reference, pre, post)
 524        self.j = float(j)
 525
 526    def initialze(self):
 527        if self.reference.weight_mode == "sparse":
 528            total_pairs = self.post.N * self.pre.N
 529            if total_pairs == 0:
 530                return
 531            rows = np.repeat(np.arange(self.post.N, dtype=np.int64), self.pre.N)

 535            return
 536        block = np.full((self.post.N, self.pre.N), self.j, dtype=self.reference.weight_dtype)

 599    def initialize(
 600        self,
 601        autapse: bool = False,
 602        weight_mode: str = "auto",
 603        ram_budget_gb: float = 12.0,
 604        weight_dtype=np.float32,
 605    ):
 606        """Allocate state, sample connectivity, and prepare the cached input field.
 607
 608        Examples
 609        --------
 610        >>> np.random.seed(1)
 611        >>> net = BinaryNetwork("init-demo")
 612        >>> pop = BinaryNeuronPopulation(net, N=2, threshold=0.1, initializer=[1, 0])
 613        >>> net.add_population(pop)
 614        <...BinaryNeuronPopulation object...>
 615        >>> net.initialize(weight_mode="dense")
 616        >>> net.field.shape
 617        (2,)
 618        """
 619        if self.N == 0:
 620            raise RuntimeError("Cannot initialize network without populations.")
 621        self.weight_dtype = np.dtype(weight_dtype)
 622        if self.weight_dtype not in (np.float32, np.float64):
 623            raise ValueError("weight_dtype must be float32 or float64.")
 624        self.state = np.zeros(self.N, dtype=np.int8)
 625        self.field = np.zeros(self.N, dtype=self.weight_dtype)
 626        self.update_prob = np.zeros(self.N, dtype=self.weight_dtype)
 627        self.population_lookup = np.zeros(self.N, dtype=np.int32)
 628        self.neuron_lookup = np.zeros(self.N, dtype=np.int32)
 629        self.thresholds = np.zeros(self.N, dtype=self.weight_dtype)
 630        pop_count = len(self.population)
 631        pop_views = np.zeros((pop_count, 2), dtype=np.int64)
 632        pop_update_mass = np.zeros(pop_count, dtype=np.float64)
 633        N_start = 0
 634        for idx, population in enumerate(self.population):
 635            population.set_view([N_start, N_start + population.N])
 636            N_start += population.N
 637            population.initialze()
 638            pop_views[idx, :] = population.view
 639            pop_update_mass[idx] = population.N / max(population.tau, 1e-9)
 640            self.update_prob[population.view[0]:population.view[1]] = 1.0 / max(population.tau, 1e-9)
 641            self.population_lookup[population.view[0]:population.view[1]] = idx
 642            self.neuron_lookup[population.view[0]:population.view[1]] = np.arange(
 643                population.N, dtype=np.int32
 644            )
 645            threshold_value = getattr(population, "threshold", 0.0)
 646            self.thresholds[population.view[0]:population.view[1]] = float(threshold_value)
 647        self.LUT = np.array([population.view for population in self.population])
 648        total = float(self.update_prob.sum())
 649        if not math.isfinite(total) or total <= 0:
 650            raise RuntimeError("Invalid update probabilities. Check tau values.")
 651        self.update_prob /= total
 652        pop_mass_total = float(pop_update_mass.sum())
 653        if not math.isfinite(pop_mass_total) or pop_mass_total <= 0:
 654            raise RuntimeError("Invalid population update masses. Check tau values.")
 655        self._population_views = pop_views
 656        self._population_cdf = np.cumsum(pop_update_mass / pop_mass_total)
 657        self.weight_mode = self._choose_weight_mode(weight_mode, ram_budget_gb)
 658        self.weights_dense = None
 659        self.weights_csr = None
 660        self.weights_csc = None
 661        self.weights = None
 662        self._sparse_rows.clear()
 663        self._sparse_cols.clear()
 664        self._sparse_data.clear()
 665        if self.weight_mode == "dense":
 666            self.weights_dense = np.zeros((self.N, self.N), dtype=self.weight_dtype)
 667        else:
 668            if sp is None:
 669                raise ModuleNotFoundError(
 670                    "SciPy is required for sparse weight mode. Install it via 'pip install scipy'."
 671                )
 672        for synapse in self.synapses:
 673            synapse.set_view(
 674                np.array(
 675                    [[synapse.pre.view[0], synapse.pre.view[1]], [synapse.post.view[0], synapse.post.view[1]]],
 676                    dtype=np.int64,
 677                )
 678            )
 679            synapse.initialze()
 680        if self.weight_mode == "dense":
 681            if not autapse:
 682                np.fill_diagonal(self.weights_dense, 0.0)
 683            self.weights = self.weights_dense
 684        else:
 685            row = np.concatenate(self._sparse_rows) if self._sparse_rows else np.zeros(0, dtype=np.int64)
 686            col = np.concatenate(self._sparse_cols) if self._sparse_cols else np.zeros(0, dtype=np.int64)
 687            data = np.concatenate(self._sparse_data) if self._sparse_data else np.zeros(0, dtype=self.weight_dtype)
 688            if not autapse and row.size:
 689                keep = row != col
 690                row = row[keep]
 691                col = col[keep]
 692                data = data[keep]
 693            matrix = sp.coo_matrix((data, (row, col)), shape=(self.N, self.N), dtype=self.weight_dtype)
 694            self.weights_csr = matrix.tocsr()
 695            self.weights_csc = matrix.tocsc()
 696            self.weights = self.weights_csr
 697        self._recompute_field()
 698        self.sim_steps = 0
 699
 700    def _recompute_field(self):
 701        state_float = self.state.astype(self.weight_dtype, copy=False)
 702        if self.weight_mode == "dense":
 703            self.field = self.weights_dense @ state_float
 704        else:
 705            self.field = self.weights_csr.dot(state_float).astype(self.weight_dtype, copy=False)

 707    def _write_weight_block(self, row_start, row_end, col_start, col_end, block):
 708        block = np.asarray(block, dtype=self.weight_dtype)
 709        if block.shape != (row_end - row_start, col_end - col_start):
 710            raise ValueError("Block shape does not match target slice.")
 711        if self.weight_mode == "dense":
 712            self.weights_dense[row_start:row_end, col_start:col_end] += block
 713        else:
 714            rows = np.arange(row_start, row_end, dtype=np.int64)
 715            cols = np.arange(col_start, col_end, dtype=np.int64)
 716            row_idx = np.repeat(rows, block.shape[1])
 717            col_idx = np.tile(cols, block.shape[0])
 718            values = block.reshape(-1)
 719            mask = values != 0.0
 720            if mask.any():
 721                self._sparse_rows.append(row_idx[mask])
 722                self._sparse_cols.append(col_idx[mask])
 723                self._sparse_data.append(values[mask])

 725    def _append_sparse_entries(self, row_idx: np.ndarray, col_idx: np.ndarray, values: np.ndarray) -> None:
 726        if self.weight_mode != "sparse":
 727            raise RuntimeError("Sparse entries can only be appended in sparse weight mode.")
 728        rows = np.asarray(row_idx, dtype=np.int64).ravel()
 729        cols = np.asarray(col_idx, dtype=np.int64).ravel()
 730        data = np.asarray(values, dtype=self.weight_dtype).ravel()
 731        if rows.size != cols.size or rows.size != data.size:
 732            raise ValueError("Sparse row/col/data arrays must have the same length.")
 733        if rows.size == 0:
 734            return
 735        mask = data != 0.0
 736        if mask.any():
 737            self._sparse_rows.append(rows[mask])
 738            self._sparse_cols.append(cols[mask])
 739            self._sparse_data.append(data[mask])

 741    def _choose_weight_mode(self, requested: str, ram_budget_gb: float) -> str:
 742        requested = str(requested or "auto").lower()
 743        if requested in {"dense", "sparse"}:
 744            return requested
 745        if requested != "auto":
 746            raise ValueError("weight_mode must be 'dense', 'sparse', or 'auto'.")
 747        # Estimate dense memory footprint assuming float32.
 748        bytes_per_entry = np.dtype(self.weight_dtype).itemsize
 749        dense_bytes = self.N * self.N * bytes_per_entry
 750        dense_gb = dense_bytes / (1024 ** 3)
 751        safety = 0.6 * float(ram_budget_gb)
 752        print("Simulating network as: " + "dense" if dense_gb <= safety else "sparse")
 753        return "dense" if dense_gb <= safety else "sparse"

 755    def _select_neurons(self, count: int) -> np.ndarray:
 756        if count <= 0:
 757            return np.zeros(0, dtype=np.int64)
 758        if self._population_cdf.size == 0:
 759            return np.random.choice(self.N, size=count, p=self.update_prob)
 760        pop_indices = np.searchsorted(self._population_cdf, np.random.random(size=count), side="right")
 761        pop_indices = np.minimum(pop_indices, self._population_views.shape[0] - 1)
 762        counts = np.bincount(pop_indices, minlength=self._population_views.shape[0])
 763        neurons = np.empty(count, dtype=np.int64)
 764        offset = 0
 765        for pop_idx, pop_count in enumerate(counts):
 766            if pop_count == 0:
 767                continue
 768            start, end = self._population_views[pop_idx]
 769            neurons[offset:offset + pop_count] = np.random.randint(start, end, size=pop_count)
 770            offset += pop_count
 771        np.random.shuffle(neurons)
 772        return neurons

 834    def _process_batch(self, neurons: Sequence[int]):
 835        neurons = np.asarray(neurons, dtype=np.int64)
 836        log_enabled = self._step_log_buffer is not None
 837        log_offset = self._step_log_index
 838        if log_enabled and self._step_log_buffer is not None:
 839            if log_offset + neurons.size > self._step_log_buffer.shape[0]:
 840                raise RuntimeError(
 841                    "Step logging buffer exhausted. Increase allocated steps when enabling step logging."
 842                )
 843            log_buffer = self._step_log_buffer
 844        else:
 845            log_buffer = self._step_log_dummy
 846        if self._diff_log_updates is not None and self._diff_log_deltas is not None:
 847            if self._diff_log_index + neurons.size > self._diff_log_updates.shape[0]:
 848                raise RuntimeError("Diff logging buffer exhausted. Increase allocated steps when enabling diff logging.")
 849        if self.weight_mode == "dense":
 850            self._update_batch_dense(neurons, log_buffer, log_enabled, log_offset)
 851        else:
 852            self._update_batch_sparse(neurons, log_buffer, log_enabled, log_offset)
 853        if log_enabled:
 854            self._step_log_index += neurons.size
 855        if self._diff_log_updates is not None and self._diff_log_deltas is not None:
 856            self._diff_log_index += neurons.size
 857        self.sim_steps += neurons.size

 859    def run(self, steps=1000, batch_size=1):
 860        """Advance the network for `steps` asynchronous updates.
 861
 862        Examples
 863        --------
 864        >>> np.random.seed(2)
 865        >>> net = BinaryNetwork("run-demo")
 866        >>> pop = BinaryNeuronPopulation(net, N=2, threshold=0.1, initializer=[1, 0])
 867        >>> net.add_population(pop)
 868        <...BinaryNeuronPopulation object...>
 869        >>> net.initialize(weight_mode="dense")
 870        >>> net.run(steps=4, batch_size=2)
 871        >>> int(net.sim_steps)
 872        4
 873        """
 874        if self.state is None or self.update_prob is None:
 875            raise RuntimeError("Call initialize() before run().")
 876        if batch_size <= 0:
 877            raise ValueError("batch_size must be positive.")
 878        steps_done = 0
 879        while steps_done < steps:
 880            current_batch = min(batch_size, steps - steps_done)
 881            neurons = self._select_neurons(current_batch)
 882            self._process_batch(neurons)
 883            steps_done += current_batch

 927    def consume_diff_log(self) -> tuple[np.ndarray, np.ndarray]:
 928        """Return the recorded diff log as `(updates, deltas)` row arrays.
 929
 930        Expected output
 931        ---------------
 932        The returned arrays have shape `(1, recorded_steps)` for the current
 933        simulation backend.
 934        """
 935        if self._diff_log_updates is None or self._diff_log_deltas is None:
 936            return np.zeros((1, 0), dtype=np.uint16), np.zeros((1, 0), dtype=np.int8)

 938        updates = self._diff_log_updates[:filled].copy()[None, :]
 939        deltas = self._diff_log_deltas[:filled].copy()[None, :]
 940        self._diff_log_updates = None
 941        self._diff_log_deltas = None
 942        self._diff_log_index = 0
 943        return updates, deltas

 947        initial_state: np.ndarray,
 948        updates: np.ndarray,
 949        deltas: np.ndarray,

 954
 955        Examples

 957        >>> BinaryNetwork.reconstruct_states_from_diff_logs(
 958        ...     initial_state=np.array([0, 0], dtype=np.uint8),

 978    def population_rates_from_diff_logs(
 979        self,
 980        initial_state: np.ndarray,
 981        updates: np.ndarray,
 982        deltas: np.ndarray,
 983        *,
 984        sample_interval: int = 1,
 985        populations: Optional[Sequence[Neuron]] = None,
 986    ) -> np.ndarray:
 987        """Compute per-population activity traces directly from diff logs.
 988
 989        Examples
 990        --------
 991        >>> net = BinaryNetwork("rates-demo")
 992        >>> pop_a = BinaryNeuronPopulation(net, N=1, initializer=[0])
 993        >>> pop_b = BinaryNeuronPopulation(net, N=1, initializer=[0])
 994        >>> net.add_population(pop_a)
 995        <...BinaryNeuronPopulation object...>
 996        >>> net.add_population(pop_b)
 997        <...BinaryNeuronPopulation object...>
 998        >>> net.initialize(weight_mode="dense")
 999        >>> rates = net.population_rates_from_diff_logs(
1000        ...     np.array([0, 0], dtype=np.uint8),
1001        ...     np.array([[0, 1, 0]], dtype=np.uint16),
1002        ...     np.array([[1, 1, -1]], dtype=np.int8),
1003        ... )
1004        >>> rates.tolist()
1005        [[1.0, 0.0], [1.0, 1.0], [0.0, 1.0]]
1006        """
1007        pops = list(self.population if populations is None else populations)
1008        if not pops:
1009            return np.zeros((0, 0), dtype=np.float32)
1010        update_arr = np.asarray(updates, dtype=np.int64)
1011        delta_arr = np.asarray(deltas, dtype=np.int8)
1012        if update_arr.ndim == 1:
1013            update_arr = update_arr[None, :]
1014        if delta_arr.ndim == 1:
1015            delta_arr = delta_arr[None, :]
1016        if update_arr.ndim != 2 or delta_arr.shape != update_arr.shape:
1017            return np.zeros((0, len(pops)), dtype=np.float32)
1018        initial = np.asarray(initial_state, dtype=np.int8).ravel()
1019        pop_count = len(pops)
1020        sizes = np.zeros(pop_count, dtype=np.float32)
1021        cluster_of = np.empty(initial.size, dtype=np.int32)
1022        cluster_sums = np.zeros(pop_count, dtype=np.int64)
1023        for idx, pop in enumerate(pops):
1024            start, end = int(pop.view[0]), int(pop.view[1])
1025            cluster_of[start:end] = idx
1026            sizes[idx] = max(1, end - start)
1027            cluster_sums[idx] = int(initial[start:end].sum())
1028        stride = max(1, int(sample_interval))
1029        return _population_rates_kernel(cluster_sums, sizes, cluster_of, update_arr, delta_arr, stride)

1031    @staticmethod
1032    def extract_spike_events_from_diff_logs(
1033        updates: np.ndarray,
1034        deltas: np.ndarray,
1035    ) -> tuple[np.ndarray, np.ndarray]:
1036        """Extract onset events `(times, neuron_ids)` from diff-log traces.
1037
1038        Examples
1039        --------
1040        >>> times, ids = BinaryNetwork.extract_spike_events_from_diff_logs(
1041        ...     np.array([[0, 1, 0]], dtype=np.uint16),
1042        ...     np.array([[1, 1, -1]], dtype=np.int8),
1043        ... )
1044        >>> times.tolist()
1045        [0.0, 1.0]
1046        >>> ids.tolist()
1047        [0, 1]
1048        """
1049        update_arr = np.asarray(updates, dtype=np.int64)
1050        delta_arr = np.asarray(deltas, dtype=np.int8)
1051        if update_arr.ndim == 1:
1052            update_arr = update_arr[None, :]
1053        if delta_arr.ndim == 1:
1054            delta_arr = delta_arr[None, :]
1055        if update_arr.ndim != 2 or delta_arr.shape != update_arr.shape:
1056            return np.zeros(0, dtype=np.float64), np.zeros(0, dtype=np.int64)
1057        mask = delta_arr > 0
1058        if not mask.any():
1059            return np.zeros(0, dtype=np.float64), np.zeros(0, dtype=np.int64)
1060        per_step, steps = update_arr.shape
1061        times = np.repeat(np.arange(steps, dtype=np.int64), per_step)
1062        flat_updates = update_arr.reshape(-1, order="F")
1063        flat_mask = mask.reshape(-1, order="F")
1064        return times[flat_mask].astype(np.float64, copy=False), flat_updates[flat_mask].astype(np.int64, copy=False)
