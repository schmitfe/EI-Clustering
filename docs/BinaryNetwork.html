BinaryNetwork

Binary network simulations for clustered E/I models.

The package contains two main layers:

  • BinaryNetwork.BinaryNetwork: generic binary-network building blocks
  • BinaryNetwork.ClusteredEI_network: repository-specific clustered E/I model assembly

Typical workflow:

import numpy as np

from BinaryNetwork import ClusteredEI_network

np.random.seed(3)
network = ClusteredEI_network(parameter)
network.initialize(weight_mode="dense")
initial_state = network.state.copy()

network.enable_diff_logging(steps=1000)
network.run(1000, batch_size=128)
updates, deltas = network.consume_diff_log()

rates = network.population_rates_from_diff_logs(
    initial_state,
    updates,
    deltas,
    sample_interval=10,
)

The recommended trace format in this repository stores:

  • initial_state
  • state_updates
  • state_deltas
  • optionally derived spike_times and spike_ids

This keeps simulations compact and allows later reconstruction of full sampled states, onset events, or population rates.

 600    def initialize(
 601        self,
 602        autapse: bool = False,
 603        weight_mode: str = "auto",

 605        weight_dtype=np.float32,

 608
 609        Examples
 610        --------
 611        >>> np.random.seed(1)
 612        >>> net = BinaryNetwork("init-demo")
 613        >>> pop = BinaryNeuronPopulation(net, N=2, threshold=0.1, initializer=[1, 0])
 614        >>> net.add_population(pop)
 615        <...BinaryNeuronPopulation object...>
 616        >>> net.initialize(weight_mode="dense")
 617        >>> net.field.shape
 618        (2,)
 619        """
 620        if self.N == 0:
 621            raise RuntimeError("Cannot initialize network without populations.")
 622        self.weight_dtype = np.dtype(weight_dtype)
 623        if self.weight_dtype not in (np.float32, np.float64):
 624            raise ValueError("weight_dtype must be float32 or float64.")
 625        self.state = np.zeros(self.N, dtype=np.int8)
 626        self.field = np.zeros(self.N, dtype=self.weight_dtype)
 627        self.update_prob = np.zeros(self.N, dtype=self.weight_dtype)
 628        self.population_lookup = np.zeros(self.N, dtype=np.int32)
 629        self.neuron_lookup = np.zeros(self.N, dtype=np.int32)
 630        self.thresholds = np.zeros(self.N, dtype=self.weight_dtype)
 631        pop_count = len(self.population)
 632        pop_views = np.zeros((pop_count, 2), dtype=np.int64)
 633        pop_update_mass = np.zeros(pop_count, dtype=np.float64)
 634        N_start = 0
 635        for idx, population in enumerate(self.population):
 636            population.set_view([N_start, N_start + population.N])
 637            N_start += population.N
 638            population.initialze()
 639            pop_views[idx, :] = population.view
 640            pop_update_mass[idx] = population.N / max(population.tau, 1e-9)
 641            self.update_prob[population.view[0]:population.view[1]] = 1.0 / max(population.tau, 1e-9)
 642            self.population_lookup[population.view[0]:population.view[1]] = idx
 643            self.neuron_lookup[population.view[0]:population.view[1]] = np.arange(
 644                population.N, dtype=np.int32
 645            )
 646            threshold_value = getattr(population, "threshold", 0.0)
 647            self.thresholds[population.view[0]:population.view[1]] = float(threshold_value)
 648        self.LUT = np.array([population.view for population in self.population])
 649        total = float(self.update_prob.sum())
 650        if not math.isfinite(total) or total <= 0:
 651            raise RuntimeError("Invalid update probabilities. Check tau values.")
 652        self.update_prob /= total
 653        pop_mass_total = float(pop_update_mass.sum())
 654        if not math.isfinite(pop_mass_total) or pop_mass_total <= 0:
 655            raise RuntimeError("Invalid population update masses. Check tau values.")
 656        self._population_views = pop_views
 657        self._population_cdf = np.cumsum(pop_update_mass / pop_mass_total)
 658        self.weight_mode = self._choose_weight_mode(weight_mode, ram_budget_gb)
 659        self.weights_dense = None
 660        self.weights_csr = None
 661        self.weights_csc = None
 662        self.weights = None
 663        self._sparse_rows.clear()
 664        self._sparse_cols.clear()
 665        self._sparse_data.clear()
 666        if self.weight_mode == "dense":
 667            self.weights_dense = np.zeros((self.N, self.N), dtype=self.weight_dtype)
 668        else:
 669            if sp is None:
 670                raise ModuleNotFoundError(
 671                    "SciPy is required for sparse weight mode. Install it via 'pip install scipy'."
 672                )
 673        for synapse in self.synapses:
 674            synapse.set_view(
 675                np.array(
 676                    [[synapse.pre.view[0], synapse.pre.view[1]], [synapse.post.view[0], synapse.post.view[1]]],
 677                    dtype=np.int64,
 678                )
 679            )
 680            synapse.initialze()
 681        if self.weight_mode == "dense":
 682            if not autapse:
 683                np.fill_diagonal(self.weights_dense, 0.0)
 684            self.weights = self.weights_dense
 685        else:
 686            row = np.concatenate(self._sparse_rows) if self._sparse_rows else np.zeros(0, dtype=np.int64)
 687            col = np.concatenate(self._sparse_cols) if self._sparse_cols else np.zeros(0, dtype=np.int64)
 688            data = np.concatenate(self._sparse_data) if self._sparse_data else np.zeros(0, dtype=self.weight_dtype)
 689            if not autapse and row.size:
 690                keep = row != col
 691                row = row[keep]
 692                col = col[keep]
 693                data = data[keep]
 694            matrix = sp.coo_matrix((data, (row, col)), shape=(self.N, self.N), dtype=self.weight_dtype)
 695            self.weights_csr = matrix.tocsr()
 696            self.weights_csc = matrix.tocsc()
 697            self.weights = self.weights_csr
 698        self._recompute_field()

 742    def _choose_weight_mode(self, requested: str, ram_budget_gb: float) -> str:
 743        requested = str(requested or "auto").lower()
 744        if requested in {"dense", "sparse"}:
 745            return requested

 747            raise ValueError("weight_mode must be 'dense', 'sparse', or 'auto'.")

 756    def _select_neurons(self, count: int) -> np.ndarray:
 757        if count <= 0:
 758            return np.zeros(0, dtype=np.int64)
 759        if self._population_cdf.size == 0:
 760            return np.random.choice(self.N, size=count, p=self.update_prob)
 761        pop_indices = np.searchsorted(self._population_cdf, np.random.random(size=count), side="right")
 762        pop_indices = np.minimum(pop_indices, self._population_views.shape[0] - 1)
 763        counts = np.bincount(pop_indices, minlength=self._population_views.shape[0])
 764        neurons = np.empty(count, dtype=np.int64)
 765        offset = 0
 766        for pop_idx, pop_count in enumerate(counts):
 767            if pop_count == 0:
 768                continue
 769            start, end = self._population_views[pop_idx]
 770            neurons[offset:offset + pop_count] = np.random.randint(start, end, size=pop_count)
 771            offset += pop_count
 772        np.random.shuffle(neurons)
 773        return neurons
